$('#inline-data').grijq({
  height: 300,
  width: 400,
  editcomplete: function(e, data) {
    $('#editcomplete').html(data.val);
  },
  rowselected: function(e, data) {
    $('#rowselected').html(data.row.text());
  },
  newrow: function() {
    var row = '<tr><td>&nbsp;</td><td></td><td></td></tr>';
    $('#inline-data').append(row);
  }
});
$('#selectLastRow').button().click(function() {
  $('#inline-data').grijq('selectLastRow');
});
$('#options input').change(function() {
  var value = this.value;
  if(this.checked) {
    value = $(this).data('unvalue');
  }
  $('#inline-data').grijq('option', this.id, value);
});
